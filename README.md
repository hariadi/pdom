# pdom


# Usage

        Usage: pdom <command> [options]

        Access the Pingdom API through a command.

          -h, --help                  Display this help message and exit
          -u, --username=STR          
          -p, --password=STR          
          -k, --key=STR               
          -v, --version               Output version information and exit


# Install

    npm -g install pdom
